    def celsius_to_fahrenheit(c: float) -> float:
        return c * 9 / 5 + 32
    
    def fahrenheit_to_celsius(f: float) -> float:
        return (f - 32) * 5 / 9
    
    if __name__ == "__main__":
        escala = input("Digite a unidade de origem (C para Celsius ou F para Fahrenheit): ").strip().upper()
        try:
            valor = float(input("Digite o valor da temperatura: ").strip())
        except ValueError:
            print("Valor numérico inválido.")
            exit(1)
    
        if escala == "C":
            convertido = celsius_to_fahrenheit(valor)
            print(f"{valor:.2f} °C equivalem a {convertido:.2f} °F")
        elif escala == "F":
            convertido = fahrenheit_to_celsius(valor)
            print(f"{valor:.2f} °F equivalem a {convertido:.2f} °C")
        else:
            print("Escala não reconhecida. Use 'C' ou 'F'.")
